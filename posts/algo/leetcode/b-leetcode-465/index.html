class Solution {
public:
    int maxN;
    bool mem[1<<20] = {0};  // é¢˜ç›®ç»™çš„æœ€å¤§å¯é€‰æ•´æ•°ä¸º20ä¸ªï¼ŒçŠ¶æ€æœ€å¤§æ˜¯20ä¸ª1ï¼Œå³2^20-1

    bool dfs(unsigned int x, int tot) {
        if(mem[x]) return true;
        for(int i=0; i<maxN; i++) {
            if(((x>>i)&1)==0) {     // å³ç§»åˆ¤æ–­ç¬¬iä¸ªæ•´æ•°æ˜¯å¦å¯é€‰
                if(i+1>=tot) {      // ç¬¬iæ•´æ•°å¯¹åº”å€¼æ˜¯i+1
                    mem[x] = true;
                    return true;
                }
                if(!dfs(x|(1<<i),tot-(i+1))) {
                    mem[x] = true;
                    return true;
                }
            }
        }
        mem[x] = false;
        return false;
    }
    bool canIWin(int maxChoosableInteger, int desiredTotal) {
        maxN = maxChoosableInteger;
        // åˆ¤æ–­ç»™å‡ºæ•´æ•°æ€»å’Œå°äºæ‰€éœ€ç´¯è®¡å’Œ
        if((1+maxN)*maxN/2<desiredTotal) {  
            return false;
        }
        return dfs(0, desiredTotal);
    }
};
