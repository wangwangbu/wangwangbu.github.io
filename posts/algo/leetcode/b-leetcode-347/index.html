// Top K using heap

type num struct {
	key int
	freq int
}

// å®ç°ä¸€ä¸ªå †æ¥å£
type myHeap []*num

func (mh myHeap) Len() int { return len(mh) }

func (mh myHeap) Less(i, j int) bool { return mh[i].freq<mh[j].freq }

func (mh myHeap) Swap(i, j int) { mh[i], mh[j] = mh[j], mh[i] }

func (mh *myHeap) Push(x interface{}) {
	xNum := x.(*num)
	*mh = append(*mh, xNum)
}

func (mh *myHeap) Pop() interface{} {
	old := *mh
	n := len(old)
	item := old[n-1]
	*mh = old[0:n-1]
	return item
}

func topKFrequent(nums []int, k int) []int {
	tMap := make(map[int]int, len(nums))
	for _, x := range nums {
		tMap[x]++;
	}
	var arr myHeap
	heap.Init(&arr)
	for key, freq := range tMap {
		if len(arr)<k {
			heap.Push(&arr, &num{key, freq})
		} else if arr[0].freq<freq {
			heap.Pop(&arr)
			heap.Push(&arr, &num{key, freq})
		}
	}
	res := []int{}
	for len(arr)>0 {
		res = append(res, heap.Pop(&arr).(*num).key)
	}
	return res
}

// Top K using hash

func topKFrequent(nums []int, k int) []int {
	tMap := make(map[int]int, len(nums))
	for _, x := range nums {
		tMap[x]++;
	}
	fMap := make(map[int]*list.List, 100005)
	max := 0    // è®°å½•æœ€å¤§é¢‘æ¬¡ max å¯ä»¥èŠ‚çœä¸€ç‚¹å¾ªç¯æ¬¡æ•°
	for key, freq := range tMap {
		ll := fMap[freq];
		if ll == nil {
			ll = list.New()
			fMap[freq] = ll
			if freq > max {
				max = freq
			}
		}
		ll.PushBack(key)
	}
	ind := 0
	res := make([]int, k)
Loop:
	for i:=max; i>=0; i-- {
		if ll, ok := fMap[i]; ok {
			for ele:=ll.Front(); ele!=nil; ele=ele.Next() {
				if ind<k {
					res[ind] = ele.Value.(int)
					ind++
				} else {
					break Loop
				}
			}
		}
	}
	return res
}

// Top K with qsort

type num struct {
	key int
	freq int
}

// éœ€è¦æä¾› start å’Œ end æ¥ç¡®å®šé€’å½’å¤„ç†çš„è¾¹ç•Œ
func findTopK(arr []*num, res []int, start, end, k, retIndex int) {
	rand.Seed(time.Now().UnixNano())
	mid := rand.Int()%(end-start+1)+start
	arr[mid], arr[start] = arr[start], arr[mid]
	midValue := (*arr[start]).freq
	index := start
	for i:=index+1; i<=end; i++ {
		if (*arr[i]).freq>midValue {
			arr[i], arr[index+1] = arr[index+1], arr[i]
			index++
		}
	}
	arr[index], arr[start] = arr[start], arr[index]
	if k<=index-start {
		findTopK(arr, res, start, index-1, k, retIndex)
	} else {
		for i:=start; i<=index; i++ {
			res[retIndex] = (*arr[i]).key
			retIndex++
		}
		if k > index-start+1 {
			findTopK(arr, res, index+1, end, k-(index-start+1), retIndex)
		}
	}
}

func topKFrequent(nums []int, k int) []int {
	tMap := make(map[int]int, len(nums))
	for _, x := range nums {
		tMap[x]++;
	}
	var arr []*num
	for k, v := range tMap {
		arr = append(arr, &num{k, v})
	}
	res := make([]int, k)
	findTopK(arr, res, 0, len(arr)-1, k, 0)
	return res
}
